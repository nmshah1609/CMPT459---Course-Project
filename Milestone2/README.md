# To run the code:
use command 'python src/prog2.py' from current directory

os, 
pandas, 
numpy, 
random, 
pickel, 
ppscore, 
sklearn, 
xgboost

## To install any library:
pip instal NAMEOFLIBRARY




eg. pip install ppscore


### To install pickle: 
pip install pickle-mixin
